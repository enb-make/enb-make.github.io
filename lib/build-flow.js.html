BuildFlow — это хэлпер для упрощения создания полностью настраиваемых технологий.

Например, сборка js:

   module.exports = require('enb/lib/build-flow').create()
       .name('js')
       .target('target', '?.js')
       .useFileList('js')
       .justJoinFilesWithComments()
       .createTech();

            getName: function () {

                return [this._target];
