¶

Осуществляет обработку файла Борщиком.

Params
sourceFilename String- Исходный файл.
destFilename String- Результирующий файл.
freeze Boolean- Осуществлять ли фризинг.
minimize Boolean- Осуществлять ли минимизацию.
[tech] String
    preprocessFile: function (sourceFilename, destFilename, freeze, minimize, tech) {
        var opts = {
            input: sourceFilename,
            output: destFilename,
            freeze: freeze,
            minimize: minimize
        };
        if (tech) {
            opts.tech = tech;
        }
        return Vow.when(borschik.api(opts));
    }
});
